    function openFullPanel(id) {
    currentEmpId = id;
    document.getElementById('f_history_body').innerHTML = '<tr><td colspan="6" class="text-center py-4">HZM VERİ TABANI SORGULANIYOR...</td></tr>';

            document.getElementById('f_name').value = data.info.name;
            document.getElementById('f_tc').value = data.info.tc_no || '';

            document.getElementById('fullUpdateForm').action = `/calisan-guncelle/${id}`;

            // --- HZM TARİH ARALIĞI OTOMASYONU ---
            // Varsayılan olarak içinde bulunduğumuz ayın ilk ve son gününü seçelim
            const today = new Date();
            const firstDay = new Date(today.getFullYear(), today.getMonth(), 1).toISOString().split('T')[0];
            const lastDay = new Date(today.getFullYear(), today.getMonth() + 1, 0).toISOString().split('T')[0];

            document.getElementById('f_wage_start').value = firstDay;
            document.getElementById('f_wage_end').value = lastDay;

            // Veriyi sakla ve tabloyu çiz
            fullHistoryData = data.history;
            renderHistoryTable(fullHistoryData);
            
            if (myModal) myModal.show();
        }).catch(err => Swal.fire('Hata', 'Veriler yüklenemedi.', 'error'));
}
    function renderHistoryTable(dataList) {
    let html = '';
    let tWork = 0, tHak = 0, tAdv = 0, tAsgariPay = 0;
    
    // Formdan ARALIK verilerini alıyoruz
    const startDate = document.getElementById('f_wage_start').value; 
    const endDate = document.getElementById('f_wage_end').value;
    const newWage = parseFloat(document.getElementById('f_wage').value || 0);
    
    dataList.forEach(h => {
        let rowAsgari = '-', rowAvans = '-', rowWork = '-', rowHakedis = '-';
        
        // --- HZM TARİH ARALIĞI HESAPLAMA MÜHÜRÜ ---
        let dailyWageAtDay;
        
        // Tarih Kontrolü: Satır tarihi seçili aralığın İÇİNDE mi?
        const isWithinRange = (startDate && endDate && h.date >= startDate && h.date <= endDate);

        if (isWithinRange && newWage > 0) {
            // Eğer aralıktaysa, formdaki yeni yevmiyeyi simüle et
            dailyWageAtDay = newWage;
        } else {
            // Aralık dışındaysa, veritabanından gelen mühürlü yevmiyeyi kullan
            // (h.daily_wage API'den gelen mühürlü değerdir)
            dailyWageAtDay = h.daily_wage || 0;
        }

        if (h.desc === 'Puantaj') {
            let workVal = parseFloat(h.work_value || 0);
            let hakVal = workVal * dailyWageAtDay; 
            
            tWork += workVal;
            tHak += hakVal;
            rowWork = workVal + ' G';
            rowHakedis = '₺' + hakVal.toLocaleString('tr-TR', {minimumFractionDigits: 2});
        } else if (h.desc === 'Ödeme 1 (Maaş)') {
            tAsgariPay += parseFloat(h.advance || 0);
            rowAsgari = '₺' + Number(h.advance).toLocaleString('tr-TR', {minimumFractionDigits: 2});
        } else {
            tAdv += parseFloat(h.advance || 0);
            rowAvans = '₺' + Number(h.advance).toLocaleString('tr-TR', {minimumFractionDigits: 2});
        }

        html += `<tr>
            <td class="font-monospace">${h.date}</td>
            <td>${rowWork}</td>
            <td class="opacity-75 font-monospace ${isWithinRange ? 'text-info fw-bold' : ''}">
                ₺${Number(dailyWageAtDay).toLocaleString('tr-TR', {minimumFractionDigits: 2})}
            </td>
            <td class="text-info fw-bold">${rowHakedis}</td>
            <td class="text-warning">${rowAsgari}</td>
            <td class="text-danger">${rowAvans}</td>
        </tr>`;
    });

    // Alt toplamları ve UI güncellemelerini yapan kısım (Mevcut kodun devamı)
    const netKalan = tHak - tAsgariPay - tAdv;
    document.getElementById('f_history_body').innerHTML = html || '<tr><td colspan="6" class="text-center py-3">Kayıt bulunamadı.</td></tr>';
    
    document.getElementById('sum_work').innerText = tWork.toFixed(2);
    document.getElementById('sum_hakedis').innerText = '₺' + tHak.toLocaleString('tr-TR', {minimumFractionDigits: 2});
    document.getElementById('sum_advance').innerText = '₺' + tAdv.toLocaleString('tr-TR', {minimumFractionDigits: 2});
    document.getElementById('sum_asgari_pay').innerText = '₺' + tAsgariPay.toLocaleString('tr-TR', {minimumFractionDigits: 2});

    const netEl = document.getElementById('sum_net');
    if(netEl) {
        netEl.innerText = '₺' + netKalan.toLocaleString('tr-TR', {minimumFractionDigits: 2});
        netEl.className = netKalan < 0 ? 'text-danger fw-bold fs-5 ms-2' : 'text-success fw-bold fs-5 ms-2';
    }
}

    // ... (Alt toplam kodların aynı kalıyor)
    const netKalan = tHak - tAsgariPay - tAdv;
    document.getElementById('f_history_body').innerHTML = html || '<tr><td colspan="6" class="text-center py-3">Kayıt bulunamadı.</td></tr>';
    
    document.getElementById('sum_work').innerText = tWork.toFixed(2);
    document.getElementById('sum_hakedis').innerText = '₺' + tHak.toLocaleString('tr-TR', {minimumFractionDigits: 2});
    document.getElementById('sum_advance').innerText = '₺' + tAdv.toLocaleString('tr-TR', {minimumFractionDigits: 2});
    document.getElementById('sum_asgari_pay').innerText = '₺' + tAsgariPay.toLocaleString('tr-TR', {minimumFractionDigits: 2});

    const netEl = document.getElementById('sum_net');
    if(netEl) {
        netEl.innerText = '₺' + netKalan.toLocaleString('tr-TR', {minimumFractionDigits: 2});
        netEl.className = netKalan < 0 ? 'text-danger fw-bold fs-5 ms-2' : 'text-success fw-bold fs-5 ms-2';
    }
    document.getElementById('summary_box').classList.remove('d-none');

    function applyFilter() {
        const start = document.getElementById('filter_start').value;
        const end = document.getElementById('filter_end').value;
        if (!start || !end) return;
        const filtered = fullHistoryData.filter(h => h.date >= start && h.date <= end);
        renderHistoryTable(filtered);
    }
